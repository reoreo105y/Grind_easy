class Solution:
    def isPalindrome(self, s: str) -> bool:
        asc, desc = [],[]

        for i in range(len(s)):
            if s[i].isalpha():
                desc.append(s[i].lower())
                asc.insert(0,s[i].lower())

        if asc == desc:
            return True
        else:
            return False
